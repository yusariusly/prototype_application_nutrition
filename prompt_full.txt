        function selectService(clickedCard) {
            // Remove selected class from all cards
            const cards = document.querySelectorAll('.service-card');
            cards.forEach(card => {
                card.classList.remove('selected');
                card.querySelector('.material-symbols-outlined.bg-white').classList.remove('opacity-100');
                card.querySelector('.material-symbols-outlined.bg-white').classList.add('opacity-0');
            });

            // Add selected class to clicked card
            clickedCard.classList.add('selected');
            clickedCard.querySelector('.material-symbols-outlined.bg-white').classList.remove('opacity-0');
            clickedCard.querySelector('.material-symbols-outlined.bg-white').classList.add('opacity-100');

            // Enable Next Button
            const nextBtn = document.getElementById('nextBtn');
            nextBtn.classList.remove('opacity-50', 'cursor-not-allowed');
            nextBtn.classList.add('hover:bg-on-primary-container', 'hover:shadow-md');
            nextBtn.disabled = false;
        }
